How to use

Browser

<script src="./dist/core-video-to-gif.min.js"></script>

Npm

$ npm install core-video-to-gif --save

In your JS File

const v2g = new CoreVideoToGif({
    // specify the video element
    el: document.querySelector('video')
})
v2g.shot({
    // options,
    start: 6, // ms
    end: 10
}, (result) => {
    // ...
    image.src = result
})
